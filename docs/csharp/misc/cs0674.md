 <span data-ttu-id="9a2e8-105">C# 컴파일러에서는 <xref:System.ParamArrayAttribute?displayProperty=nameWithType>를 사용할 수 없습니다. 대신 [params](../../csharp/language-reference/keywords/params.md) 를 사용하세요.</span><span class="sxs-lookup"><span data-stu-id="9a2e8-105">The C# compiler does not allow for the use of <xref:System.ParamArrayAttribute?displayProperty=nameWithType>; use [params](../../csharp/language-reference/keywords/params.md) instead.</span></span>  
  
 <span data-ttu-id="9a2e8-106">다음 샘플에서는 CS0674를 생성합니다.</span><span class="sxs-lookup"><span data-stu-id="9a2e8-106">The following sample generates CS0674:</span></span>  
  
```csharp  
// CS0674.cs  
using System;  
public class MyClass   
{  
  
   public static void UseParams([ParamArray] int[] list)   // CS0674  
   // try the following line instead  
   // public static void UseParams(params int[] list)   
   {  
      for ( int i = 0 ; i < list.Length ; i++ )  
         Console.WriteLine(list[i]);  
      Console.WriteLine();  
   }  
  
   public static void Main()   
   {  
      UseParams(1, 2, 3);  
   }  
}  
```

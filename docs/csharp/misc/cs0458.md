 <span data-ttu-id="808d9-107">이 예제에서는 이 오류를 발생시키는 nullable 값 형식을 사용하여 여러 가지 작업을 보여 줍니다.</span><span class="sxs-lookup"><span data-stu-id="808d9-107">This example illustrates a number of the different operations with nullable value types that will cause this error.</span></span>

using System;  
public  class Test
{  
    public static void Main()  
    {  
        int a = 5;  
        int? b = a + null;    // CS0458  
        int? qa = 15;  
        b = qa + null;        // CS0458  
        b -= null;            // CS0458  
        int? qa2 = null;  
        b = qa2 + null;       // CS0458  
        qa2 -= null;          // CS0458  
    }  
}

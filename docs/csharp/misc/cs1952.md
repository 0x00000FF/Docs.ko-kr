## <a name="example"></a><span data-ttu-id="dc9f9-107">예제</span><span class="sxs-lookup"><span data-stu-id="dc9f9-107">Example</span></span>  
 <span data-ttu-id="dc9f9-108">다음 코드는 CS1952를 생성합니다.</span><span class="sxs-lookup"><span data-stu-id="dc9f9-108">The following code produces CS1952:</span></span>

using System;  
using System.Linq.Expressions;  
  
class Test  
{  
    public static int M(__arglist)  
    {  
        return 1;  
    }  
  
    static int Main()  
    {  
        Expression<Func<int, int>> f = x => Test.M(__arglist(x)); // CS1952  
        return 1;  
    }  
}

## <a name="example"></a><span data-ttu-id="a5ec0-108">예제</span><span class="sxs-lookup"><span data-stu-id="a5ec0-108">Example</span></span>

class CS1917  
{  
    public struct TestStruct  
    {  
        public int i;  
    }  
    public class C  
    {  
        public readonly TestStruct str = new TestStruct();  
        public static int Main()  
        {  
            C c = new C { str = { i = 1 } }; // CS1917  
            return 0;  
        }  
    }  
}

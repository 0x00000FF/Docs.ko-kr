## <a name="example"></a><span data-ttu-id="6167f-105">예제</span><span class="sxs-lookup"><span data-stu-id="6167f-105">Example</span></span>

```csharp  
// CS0724.cs  
using System;  
  
class X  
{  
    static void Test()  
    {  
        try  
        {  
            throw new Exception();  
        }  
        catch  
        {  
            try  
            {  
            }  
            finally  
            {  
                throw; // CS0724  
            }  
        }  
    }  
  
    static void Main()  
    {  
    }  
}  
```

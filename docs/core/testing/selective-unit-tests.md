## <a name="mstest"></a><span data-ttu-id="48768-107">MSTest</span><span class="sxs-lookup"><span data-stu-id="48768-107">MSTest</span></span>

```csharp

{
    using Microsoft.VisualStudio.TestTools.UnitTesting;

    [TestClass]
    public class UnitTestClass1
    {
        [Priority(2)]
        [TestMethod]
        public void TestMethod1()
        {
        }

        [TestCategory("CategoryA")]
        [Priority(3)]
        [TestMethod]
        public void TestMethod2()
        {
        }
    }
}
```
